手写 bind

发布日期:   2021-04-15

常见的调用方式如下

func.bind(obj);

bind是绑定函数执行上下文中的this属性，bind和call，apply其他两个函数不一样，它不只是一次函数调用，它还返回了一个新的函数，所以bind的基本结构如下

function bind(thisArg) {

  return function() {
    // ...
  };
}

由于是在func函数上调用的bind，因此bind是绑定到Function.prototype上的，于是

Function.prototype.bind = function bind(thisArg) {
  return function() {
    // ...
  }
}

基本格式确定了，关键是怎么把thisArg设置为新函数的this呢？假如可以使用apply的话，只要获取到之前的函数，然后调用apply绑定新的this即可。

Function.prototype.bind = function bind(thisArg) {
  const self = this;
  return function() {
    self.apply(thisArg)
    // ...
  }
}

进一步，我们需要考虑bind是可以预置参数的，而之后新函数调用也是可以传递参数的，对参数进行一个合并

Function.prototype.bind = function bind(thisArg, ...bindArgs) {
  const self = this;
  return function(...args) {
    const newArgs = bindArgs.concat(args);
    self.apply(thisArg, newArgs);
  }
}

如果是构造函数使用上面定义的bind，自然是不行的，因为很明显 bind 之后返回了新的函数，而非原来的构造函数。如下所示

Function.prototype.mybind = function name(thisArg, ...bindArgs) {
  const self = this;
  return function(...args) {
    const newArgs = bindArgs.concat(args);
    self.apply(thisArg, newArgs);
  }
}

function Point (x, y) {
  this.x = x;
  this.y = y;
}

Point.prototype.toString = function() {
  return this.x + ',' + this.y;
};

// 使用 mybind
let YAxisPoint = Point.mybind(null, 0);
let p1 = new YAxisPoint(5);
console.log(p1.toString()) // [object Object]
console.log(p1.__proto__ === Point.prototype); // false

// 使用原生bind
let YAxisPoint2 = Point.bind(null, 0);
let p2 = new YAxisPoint2(5);
console.log(p2.toString()) // 0,5
console.log(p2.__proto__ === Point.prototype); // true

对于YAxisPoint来说，它是在mybind里面返回的那个函数，并不是Point或者Point的子类，因而p1.toString的结果并非预期的是0,5。

如果要与原生的bind一样，支持对新的函数使用new进行实例化，那么其实就是要实现一个”经典继承“，或者说”盗用构造函数“，因此对于mybind函数而言，就是要同时支持直接调用或实现经典继承，关键一步的伪代码如下

self.apply(isCallByNew ? this : thisArg, newArgs);

其实就是this绑定目标对象的区别。当然，以上是不够的，还需要继承原构造函数的prototype

Function.prototype.mybind = function name(thisArg, ...bindArgs) {
  const self = this;
  function func(...args) {
    const newArgs = bindArgs.concat(args);
    self.apply(this instanceof func ? this : thisArg, newArgs);
  }

  if (self.prototype) {
    func.prototype = self.prototype;
  }

  return func;
}

这里设置了新的构造函数func和原构造函数的prototype指向同一个。

最终的代码如下

Function.prototype.mybind = function name(thisArg, ...bindArgs) {
  const self = this;
  function func(...args) {
    const newArgs = bindArgs.concat(args);
    self.apply(this instanceof func ? this : thisArg, newArgs);
  }

  if (self.prototype) {
    func.prototype = self.prototype;
  }

  return func;
}

function Point (x, y) {
  this.x = x;
  this.y = y;
}

Point.prototype.toString = function() {
  return this.x + ',' + this.y;
};

// 使用 mybind
let YAxisPoint = Point.mybind(null, 0);
let p1 = new YAxisPoint(5);
console.log(p1.toString()) // 0,5
console.log(p1.__proto__ === Point.prototype); // true

有的实现会考虑新老语法的兼容，我觉得大可不必，写 bind 的目的并不在此。

有的实现考虑使用原型链来生成新的prototype，我觉得也大可不必，要实现的是 bind，并不是要学习所谓的组合继承。
